60  void fill_lj(LibRaw_byte_buffer* buf,int nbits)
61  {
62  unsigned c1,c2,c3;
63  if(rst || nbits < vbits) return;
64  int m = vbits >> 3;
65  switch(m)
66  {
67  case 2:
68  c1 = buf->get_ljpeg_byte();
69  bitbuf = (bitbuf <<8) | (c1);
70  vbits+=8;
71  break;
72  case 1:
73  c1 = buf->get_ljpeg_byte();
74  c2 = buf->get_ljpeg_byte();
75  bitbuf = (bitbuf <<16) | (c1<<8) | c2;
76  vbits+=16;
77  break;
78  case 0:
79  c1 = buf->get_ljpeg_byte();
80  c2 = buf->get_ljpeg_byte();
81  c3 = buf->get_ljpeg_byte();
82  bitbuf = (bitbuf <<24) | (c1<<16) | (c2<<8)|c3;
83  vbits+=24;
84  break;
85  }
