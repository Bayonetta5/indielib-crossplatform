359 struct structMatrix {
360  //Matrix (OpenGL COLUM-MAJOR ORDER!) (Note indices are the same, just store order in memory is different)
361  //Be aware that DirectX matrixes are stored in memory using ROW-MAJOR ORDER!. Use methods here to convert
362  //between them safely.
363  float _11, _21, _31, _41;
364  float _12, _22, _32, _42;
365  float _13, _23, _33, _43;
366  float _14, _24, _34, _44;
367 
368  structMatrix() {
369  _11 = _12 = _13 = _14 = _21 = _22 = _23 = _24 = _31 = _32 = _33 = _34 = _41 = _42 = _43 = _44 = 0;
370  }
371  //Initializes the structure with a 4x4 matrix in array format.
372  //Elements are interpreted this way:
373  //Matrix 4x4\n (_11, _12, _13, _14,\n _21, _22, _23, _24,\n _31, _32, _33, _34\n _41, _42, _43, _44)
374  structMatrix(float* matrixArray){
375  if (!matrixArray) {
376  _11 = _12 = _13 = _14 = _21 = _22 = _23 = _24 = _31 = _32 = _33 = _34 = _41 = _42 = _43 = _44 = 0;
377  return;
378  }
379  readFromArray(matrixArray);
380  }
381 
382  //Writes to passed matrixArray the values inside the IND_Matrix structure
383  //Elements are interpreted this way:
384  //Matrix 4x4\n (_11, _12, _13, _14,\n _21, _22, _23, _24,\n _31, _32, _33, _34\n _41, _42, _43, _44)
385  void arrayRepresentation(float* matrixArray) {
386  if (!matrixArray) {
387  return;
388  }
389  matrixArray[0] = _11;
390  matrixArray[1] = _21;
391  matrixArray[2] = _31;
392  matrixArray[3] = _41;
393  matrixArray[4] = _12;
394  matrixArray[5] = _22;
395  matrixArray[6] = _32;
396  matrixArray[7] = _42;
397  matrixArray[8] = _13;
398  matrixArray[9] = _23;
399  matrixArray[10] = _33;
400  matrixArray[11] = _43;
401  matrixArray[12] = _14;
402  matrixArray[13] = _24;
403  matrixArray[14] = _34;
404  matrixArray[15] = _44;
405  }
406 
407  structMatrix & operator=(const structMatrix &rhs) {
408  if (this != &rhs) {
409  _11 = rhs._11;
410  _21 = rhs._21;
411  _31 = rhs._31;
412  _41 = rhs._41;
413  _12 = rhs._12;
414  _22 = rhs._22;
415  _32 = rhs._32;
416  _42 = rhs._42;
417  _13 = rhs._13;
418  _23 = rhs._23;
419  _33 = rhs._33;
420  _43 = rhs._43;
421  _14 = rhs._14;
422  _24 = rhs._24;
423  _34 = rhs._34;
424  _44 = rhs._44;
425  }
426  return *this;
427  }

431  //Matrix 4x4\n (_11, _12, _13, _14,\n _21, _22, _23, _24,\n _31, _32, _33, _34\n _41, _42, _43, _44)

433  _11 = matrixArray[0];

444  _43 = matrixArray[11];
445  _14 = matrixArray[12];
446  _24 = matrixArray[13];
447  _34 = matrixArray[14];
448  _44 = matrixArray[15];
449  }
