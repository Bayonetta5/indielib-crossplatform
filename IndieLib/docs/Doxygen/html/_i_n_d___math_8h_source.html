525  bool isCircleToCircleCollision(IND_Vector2 &pP1,
526  int pRadius1,
527  IND_Vector2 &pP2,
528  int pRadius2) {
529  // h^2 = x^2 + y^2 Pythagoras :D
530 
531  // x^2
532  double mDeltaXSquared = pP1._x - pP2._x;
533  mDeltaXSquared *= mDeltaXSquared;
534 
535  // y^2
536  double mDeltaYSquared = pP1._y - pP2._y;
537  mDeltaYSquared *= mDeltaYSquared;
538 
539  // Adding radius^2
540  double mSumRadiiSquared = pRadius1 + pRadius2;
541  mSumRadiiSquared *= mSumRadiiSquared;
542 
543  // If that result is greater than h^2, so less than x^2 + y^2
544  // then there is a collision
545  if (mDeltaXSquared + mDeltaYSquared <= mSumRadiiSquared)
546  return 1;
547  else
548  return 0;

587  static bool isPointInsideTriangle(IND_Vector2 &p,
588  IND_Vector2 &a,
589  IND_Vector2 &b,
590  IND_Vector2 &c) {
591  // Compute vectors
592  IND_Vector2 v0 = c - a;
593  IND_Vector2 v1 = b - a;
594  IND_Vector2 v2 = p - a;
595 
596  // Compute dot products
597  float dot00 = v0.dotProduct(v0);
598  float dot01 = v0.dotProduct(v1);
599  float dot02 = v0.dotProduct(v2);
600  float dot11 = v1.dotProduct(v1);
601  float dot12 = v1.dotProduct(v2);
602 
603  // Compute barycentric coordinates
604  float invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
605  float u = (dot11 * dot02 - dot01 * dot12) * invDenom;
606  float v = (dot00 * dot12 - dot01 * dot02) * invDenom;
607 
608  // Check if point is in triangle
609  return (u > 0) && (v > 0) && (u + v < 1);
610  }

621  double pointToLineDistance(IND_Vector2 &pA,
622  IND_Vector2 &pB,
623  IND_Vector2 &pC,
624  bool pIsSegment) {
625  IND_Vector2 ab (pB._x - pA._x, pB._y - pA._y);
626  IND_Vector2 ac (pC._x - pA._x, pC._y - pA._y);
627 
628  double mDist (ab.crossProduct(ac) / pA.distance(pB));
629 
630  if (pIsSegment) {
631  IND_Vector2 bc (pC._x - pB._x, pC._y - pB._y);
632  IND_Vector2 ba (pA._x - pB._x, pA._y - pB._y);
633 
634  float mDot1 (ab.dotProduct(bc));
635  if (mDot1 > 0.0f) {
636  return pB.distance(pC);
637  }
638  float mDot2 (ba.dotProduct(ac));
639  if (mDot2 > 0.0f) {
640  return pA.distance(pC);
641  }
642  }
643 
644  return fabs(mDist);
645  }

657  bool isSegmentIntersection(IND_Vector2 &a,
658  IND_Vector2 &b,
659  IND_Vector2 &c,
660  IND_Vector2 &d) {
661  float mGradAB, mGradCD, mYcptAB, mYcptCD, mInterceptX, mInterceptY;
662 
663  // In order to avoid divisions by zero
664  if (a._y == b._y)
665  b._y += 0.0001f;
666 
667  if (a._x == b._x)
668  b._x += 0.0001f;
669 
670  if (c._y == d._y)
671  d._y += 0.0001f;
672 
673  if (c._x == d._x)
674  d._x += 0.0001f;
675 
676  // Calculates the intersection between the two lines
677  mGradAB = (a._y - b._y) / (a._x - b._x);
678  mGradCD = (c._y - d._y) / (c._x - d._x);
679 
680  mYcptAB = a._y - a._x * mGradAB;
681  mYcptCD = c._y - c._x * mGradCD;
682  mInterceptX = (mYcptAB - mYcptCD) / (mGradCD - mGradAB);
683  mInterceptY = (mYcptAB - (mGradAB * mYcptCD) / mGradCD) / (1 - mGradAB / mGradCD);
684 
685  // Checking in the intersection is inside the segment
686  if (!((mInterceptX >= a._x && mInterceptX <= b._x) || (mInterceptX >= b._x && mInterceptX <= a._x)))
687  return 0;
688 
689  if (!((mInterceptY >= a._y && mInterceptY <= b._y) || (mInterceptY >= b._y && mInterceptY <= a._y)))
690  return 0;
691 
692  if (!((mInterceptX >= c._x && mInterceptX <= d._x) || (mInterceptX >= d._x && mInterceptX <= c._x)))
693  return 0;
694 
695  if (!((mInterceptY >= c._y && mInterceptY <= d._y) || (mInterceptY >= d._y && mInterceptY <= c._y)))
696  return 0;
697 
698  return 1;
699  }
