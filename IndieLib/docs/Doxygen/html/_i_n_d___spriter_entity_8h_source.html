60  // it orders them consistently.

62  if (folderId < other.folderId) return true; else
63  if (folderId == other.folderId) {
64  if (fileId < other.fileId) return true; else
65  if (fileId == other.fileId) {
66  // We should never reach this point
67 
68  }
69  }
70 
71  return false;
72  }
